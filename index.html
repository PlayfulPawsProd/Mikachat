// sw.js - Minimal Service Worker for PWA Installability self.addEventListener('install', (event) => { console.log('Mika SW installed!', event); self.skipWaiting(); // Activate new SW immediately }); self.addEventListener('activate', (event) => { console.log('Mika SW activated!', event); event.waitUntil(self.clients.claim()); // Claim clients immediately }); self.addEventListener('fetch', (event) => { // Minimal fetch handler just passes through requests event.respondWith(fetch(event.request)); });
